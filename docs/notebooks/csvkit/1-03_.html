        state VARCHAR NOT NULL,
        county VARCHAR NOT NULL,
        fips FLOAT NOT NULL,
        nsn VARCHAR NOT NULL,
        item_name VARCHAR,
        quantity FLOAT NOT NULL,
        ui VARCHAR NOT NULL,
        acquisition_cost FLOAT NOT NULL,

        federal_supply_category FLOAT NOT NULL,

        federal_supply_class FLOAT NOT NULL,

);
